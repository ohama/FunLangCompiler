// Then block: detect if thenOps ends with terminator (nested match/if)
let thenBlockBody =
    let isTerminator op =
        match op with
        | CfBrOp _ | CfCondBrOp _ | LlvmUnreachableOp -> true
        | _ -> false
    match List.tryLast thenOps with
    | Some op when isTerminator op && blocksAfterThen > blocksBeforeThen ->
        // thenExpr created side blocks (match inside then).
        // Patch CfBrOp into the match's merge block (last side block from thenExpr).
