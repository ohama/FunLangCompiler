```
bash -c 'PROJROOT=%S/../.. && D=$(mktemp -d /tmp/fncbuild_XXXXXX) && ln -s "$PROJROOT/Prelude" "$D/Prelude" && printf "[project]\nname = \"test\"\n\n[[executable]]\nname = \"hello\"\nmain = \"hello.fun\"\n" > "$D/funproj.toml" && printf "let _ = println \"hello from build\"\n" > "$D/hello.fun" && cd "$D" && dotnet run --project "$PROJROOT/src/FunLangCompiler.Cli/FunLangCompiler.Cli.fsproj" -- build 2>&1 && ./build/hello; EC=$?; rm -rf "$D"; exit $EC'
```
