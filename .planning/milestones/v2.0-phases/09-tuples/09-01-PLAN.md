**2. Add `Tuple` elaboration case** (before the `| _ ->` catch-all in `elaborateExpr`):

```fsharp
| Tuple (exprs, _) ->
    let n = List.length exprs
    // Elaborate all field expressions first
    let fieldResults = exprs |> List.map (fun e -> elaborateExpr env e)
    let allFieldOps  = fieldResults |> List.collect snd
    let fieldVals    = fieldResults |> List.map fst
    // GC_malloc(n * 8) to allocate the tuple struct
    let bytesVal  = { Name = freshName env; Type = I64 }
    let tupPtrVal = { Name = freshName env; Type = Ptr }
    let allocOps  = [
        ArithConstantOp(bytesVal, int64 (n * 8))
        LlvmCallOp(tupPtrVal, "@GC_malloc", [bytesVal])
    ]
    // Store each field: GEP ptr[i] + store value at slot
    let storeOps =
        fieldVals |> List.mapi (fun i fv ->
            let slotVal = { Name = freshName env; Type = Ptr }
            [ LlvmGEPLinearOp(slotVal, tupPtrVal, i)
              LlvmStoreOp(fv, slotVal) ]
        ) |> List.concat
    (tupPtrVal, allFieldOps @ allocOps @ storeOps)
```

**3. Add `LetPat(TuplePat(...))` elaboration case** (immediately after the existing `LetPat(VarPat(...))` case around line 260):

```fsharp
| LetPat (TuplePat (pats, _), bindExpr, bodyExpr, _) ->
    // 1. Elaborate the tuple expression → tupPtrVal : Ptr
    let (tupPtrVal, bindOps) = elaborateExpr env bindExpr
    // Helper: determine load type from sub-pattern shape
    let loadTypeOfPat = function
        | TuplePat _ -> Ptr
        | _          -> I64
    // 2. GEP + load for each sub-pattern field
    let (extractOps, env') =
        pats
        |> List.mapi (fun i pat -> (i, pat))
        |> List.fold (fun (opsAcc, envAcc: ElabEnv) (i, pat) ->
            let slotVal  = { Name = freshName env; Type = Ptr }
            let fieldVal = { Name = freshName env; Type = loadTypeOfPat pat }
            let gepOp  = LlvmGEPLinearOp(slotVal, tupPtrVal, i)
            let loadOp = LlvmLoadOp(fieldVal, slotVal)
            let envAcc' =
                match pat with
                | VarPat (name, _) ->
                    { envAcc with Vars = Map.add name fieldVal envAcc.Vars }
                | WildcardPat _ ->
                    envAcc
                | TuplePat _ ->
                    // Nested tuple: bind a temp name for further destructuring
                    // The inner LetPat will use this value via a fresh temp binding
                    // For now expose as anonymous — the body must use another LetPat
                    envAcc
                | _ ->
                    failwithf "Elaboration: unsupported sub-pattern in TuplePat: %A" pat
            (opsAcc @ [gepOp; loadOp], envAcc')
        ) ([], env)
    // 3. Elaborate the body with the extended environment
    let (bodyVal, bodyOps) = elaborateExpr env' bodyExpr
    (bodyVal, bindOps @ extractOps @ bodyOps)
```

Wait — nested `TuplePat` sub-patterns need the inner ptr to be accessible. Revise: when a sub-pattern is `TuplePat`, bind the loaded `Ptr` to a generated temp name so the outer LetPat body can be `LetPat(TuplePat..., Var(tempName), ...)`. BUT: FunLang uses `let (x, (y, z)) = expr` which the parser emits as a single `LetPat(TuplePat([VarPat "x"; TuplePat([VarPat "y"; VarPat "z"])]))`. So we need to handle nested `TuplePat` recursively.

**Revised approach — use a recursive helper `bindPattern`:**

```fsharp
| LetPat (TuplePat (pats, _), bindExpr, bodyExpr, _) ->
    let (tupPtrVal, bindOps) = elaborateExpr env bindExpr
    // Recursively bind patterns to extracted values from a tuple ptr
    let rec bindTuplePat (env: ElabEnv) (ptrVal: MlirValue) (pats: Pattern list) : MlirOp list * ElabEnv =
        let loadTypeOfPat = function
            | TuplePat _ -> Ptr
            | _          -> I64
        pats
        |> List.mapi (fun i pat -> (i, pat))
        |> List.fold (fun (opsAcc, envAcc: ElabEnv) (i, pat) ->
            let slotVal  = { Name = freshName env; Type = Ptr }
            let fieldVal = { Name = freshName env; Type = loadTypeOfPat pat }
            let gepOp  = LlvmGEPLinearOp(slotVal, ptrVal, i)
            let loadOp = LlvmLoadOp(fieldVal, slotVal)
            match pat with
            | VarPat (name, _) ->
                let envAcc' = { envAcc with Vars = Map.add name fieldVal envAcc.Vars }
                (opsAcc @ [gepOp; loadOp], envAcc')
            | WildcardPat _ ->
                (opsAcc @ [gepOp; loadOp], envAcc)
            | TuplePat (innerPats, _) ->
                // fieldVal is a Ptr to the inner tuple — recurse
                let (innerOps, innerEnv) = bindTuplePat envAcc fieldVal innerPats
                (opsAcc @ [gepOp; loadOp] @ innerOps, innerEnv)
            | _ ->
                failwithf "Elaboration: unsupported sub-pattern in TuplePat: %A" pat
        ) ([], env)
    let (extractOps, env') = bindTuplePat env tupPtrVal pats
    let (bodyVal, bodyOps) = elaborateExpr env' bodyExpr
    (bodyVal, bindOps @ extractOps @ bodyOps)
```
