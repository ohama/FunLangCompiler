Replace that entire `| _` catch-all with these four cases:

```fsharp
    | ConstPat(StringConst s, _) ->
        // Elaborate the string literal for the pattern constant
        let (patStrVal, patStrOps) = elaborateStringLiteral env s
        // Get data ptr for pattern string: GEP field 1
        let patDataPtrVal  = { Name = freshName env; Type = Ptr }
        let patDataVal     = { Name = freshName env; Type = Ptr }
        // Get data ptr for scrutinee string: GEP field 1
        let scrutDataPtrVal = { Name = freshName env; Type = Ptr }
        let scrutDataVal    = { Name = freshName env; Type = Ptr }
        // strcmp result
        let cmpRes  = { Name = freshName env; Type = I32 }
        let zero32  = { Name = freshName env; Type = I32 }
        let cond    = { Name = freshName env; Type = I1 }
        let ops = patStrOps @ [
            LlvmGEPStructOp(patDataPtrVal, patStrVal, 1)
            LlvmLoadOp(patDataVal, patDataPtrVal)
            LlvmGEPStructOp(scrutDataPtrVal, scrutVal, 1)
            LlvmLoadOp(scrutDataVal, scrutDataPtrVal)
            LlvmCallOp(cmpRes, "@strcmp", [scrutDataVal; patDataVal])
            ArithConstantOp(zero32, 0L)
            ArithCmpIOp(cond, "eq", cmpRes, zero32)
        ]
        (Some cond, ops, env)

    | EmptyListPat _ ->
        // Null check: scrutVal == null means empty list
        let nullVal = { Name = freshName env; Type = Ptr }
        let cond    = { Name = freshName env; Type = I1 }
        let ops     = [ LlvmNullOp(nullVal); LlvmIcmpOp(cond, "eq", scrutVal, nullVal) ]
        (Some cond, ops, env)

    | ConsPat(hPat, tPat, _) ->
        // Non-null check: scrutVal != null means cons cell
        let nullVal   = { Name = freshName env; Type = Ptr }
        let isNonNull = { Name = freshName env; Type = I1 }
        // Load head at slot 0 (base ptr — no GEP needed)
        let headVal  = { Name = freshName env; Type = I64 }
        // Load tail at slot 1
        let tailSlot = { Name = freshName env; Type = Ptr }
        let tailVal  = { Name = freshName env; Type = Ptr }
        let ops = [
            LlvmNullOp(nullVal)
            LlvmIcmpOp(isNonNull, "ne", scrutVal, nullVal)
            LlvmLoadOp(headVal, scrutVal)               // head at slot 0
            LlvmGEPLinearOp(tailSlot, scrutVal, 1)      // &cell.tail
            LlvmLoadOp(tailVal, tailSlot)               // tail pointer
        ]
        // Bind sub-patterns
        let env' =
            match hPat with
            | VarPat(n, _) -> { env with Vars = Map.add n headVal env.Vars }
            | WildcardPat _ -> env
            | _ -> failwithf "testPattern: ConsPat head sub-pattern must be VarPat or WildcardPat (got %A)" hPat
        let env'' =
            match tPat with
            | VarPat(n, _) -> { env' with Vars = Map.add n tailVal env'.Vars }
            | WildcardPat _ -> env'
            | _ -> failwithf "testPattern: ConsPat tail sub-pattern must be VarPat or WildcardPat (got %A)" tPat
        (Some isNonNull, ops, env'')

    | TuplePat(pats, _) ->
        // Tuple always matches structurally — unconditional.
        // GEP each field, load with appropriate type, bind sub-patterns.
        let loadTypeOfPat = function
            | TuplePat _ -> Ptr
            | _          -> I64
        let (ops, bindEnv) =
            pats
            |> List.mapi (fun i pat -> (i, pat))
            |> List.fold (fun (opsAcc, eAcc: ElabEnv) (i, subPat) ->
                let slotVal  = { Name = freshName env; Type = Ptr }
                let fieldVal = { Name = freshName env; Type = loadTypeOfPat subPat }
                let gepOp    = LlvmGEPLinearOp(slotVal, scrutVal, i)
                let loadOp   = LlvmLoadOp(fieldVal, slotVal)
                match subPat with
                | VarPat(vname, _) ->
                    let eAcc' = { eAcc with Vars = Map.add vname fieldVal eAcc.Vars }
                    (opsAcc @ [gepOp; loadOp], eAcc')
                | WildcardPat _ ->
                    (opsAcc @ [gepOp; loadOp], eAcc)
                | TuplePat _ ->
                    // Nested tuple — recurse: fieldVal is a Ptr to the inner tuple
                    // The inner TuplePat will be handled when testPattern is called recursively
                    // For now, bind nothing at this level; caller must recurse
                    // Actually: flatten by recursively calling testPattern on the inner tuple
                    // with fieldVal as the new scrutinee
                    let (innerCond, innerOps, innerEnv) = testPattern eAcc fieldVal subPat
                    // innerCond should be None (TuplePat is unconditional)
                    ignore innerCond
                    (opsAcc @ [gepOp; loadOp] @ innerOps, innerEnv)
                | _ ->
                    failwithf "testPattern: TuplePat sub-pattern %A not supported" subPat
            ) ([], env)
        (None, ops, bindEnv)

    | _ ->
        failwithf "testPattern: pattern %A not supported in v2" pat
```

IMPORTANT constraints:
- `elaborateStringLiteral` is already defined in Elaboration.fs (used by Phase 8). Call it directly.
- `LlvmGEPStructOp` uses `!llvm.struct<(i64, ptr)>` type — this is the string header struct layout.
- `LlvmGEPLinearOp` uses linear byte-offset GEP — this is the tuple/cons cell slot access.
- For `ConsPat`, head is hardcoded `I64` (integer list scope in v2). Tail is always `Ptr`.
- For `TuplePat`, `loadTypeOfPat` determines whether to load as `Ptr` (nested tuple) or `I64` (leaf value). This mirrors the Phase 9 `bindTuplePat` heuristic in `LetPat(TuplePat)` case.
- `@strcmp` is already declared in ExternalFuncs unconditionally. `@GC_malloc` is also present (needed by `elaborateStringLiteral`).
