**3. Add ensureRecordFieldTypes (after ensureAdtFieldTypes, around line 1080):**

This is the CRITICAL function. The MatchCompiler generates `argAccessors = [Field(scrutAcc, 0), Field(scrutAcc, 1), ...]` in sorted (alphabetical) field name order. But the heap slots use declaration order from RecordEnv. When these differ (e.g., `type T = { y: int; x: int }` -- y at slot 0, x at slot 1, but alphabetically x comes first), we must override the accessor cache to load from the correct declaration-order slot.

```fsharp
// Pre-populate accessor cache for RecordCtor sub-fields with correct declaration-order slots.
// CRITICAL: argAccs.[i] = Field(scrutAcc, i) uses alphabetical-order index i,
// but the heap layout uses declaration-order slot from RecordEnv. We must load
// each field from its actual declaration-order slot and cache it at the argAccs key.
let ensureRecordFieldTypes (fields: string list) (scrutAcc: MatchCompiler.Accessor) (argAccs: MatchCompiler.Accessor list) : MlirOp list =
    // fields is the sorted (alphabetical) field name list from RecordCtor
    let fieldSet = Set.ofList fields
    let fieldMap =
        env.RecordEnv
        |> Map.toSeq
        |> Seq.tryPick (fun (_, fm) ->
            let fmFields = fm |> Map.toSeq |> Seq.map fst |> Set.ofSeq
            if fieldSet |> Set.forall (fun f -> Set.contains f fmFields) then Some fm
            else None)
        |> Option.defaultWith (fun () ->
            failwithf "ensureRecordFieldTypes: cannot resolve record type for fields %A" fields)
    let mutable ops = []
    fields |> List.iteri (fun i fieldName ->
        if i < argAccs.Length then
            let declSlotIdx = Map.find fieldName fieldMap  // declaration-order slot
            // Resolve the scrutinee (parent) as Ptr
            let (parentVal, parentOps) = resolveAccessorTyped scrutAcc Ptr
            let slotPtr  = { Name = freshName env; Type = Ptr }
            let fieldVal = { Name = freshName env; Type = I64 }
            let gepOp  = LlvmGEPLinearOp(slotPtr, parentVal, declSlotIdx)
            let loadOp = LlvmLoadOp(fieldVal, slotPtr)
            accessorCache.[argAccs.[i]] <- fieldVal
            ops <- ops @ parentOps @ [gepOp; loadOp]
    )
    ops
```
