Modify `src/FunLangCompiler.Cli/Program.fs` to load and prepend Prelude .fun files before parsing.

In the `main` function, after reading `src` from the input file and before `let ast = parseProgram src inputPath`, add prelude loading logic:

```fsharp
// After: let src = System.IO.File.ReadAllText(inputPath)
// Before: let ast = parseProgram src inputPath

// Phase 35: Auto-load Prelude modules
let preludeSrc =
    let findPreludeDir () =
        // 1. Check CWD/Prelude
        let cwdCandidate = "Prelude"
        if System.IO.Directory.Exists cwdCandidate then cwdCandidate
        else
            // 2. Check assembly directory/Prelude
            let asmDir = System.IO.Path.GetDirectoryName(System.Reflection.Assembly.GetEntryAssembly().Location)
            let asmCandidate = System.IO.Path.Combine(asmDir, "Prelude")
            if System.IO.Directory.Exists asmCandidate then asmCandidate
            else ""
    let dir = findPreludeDir ()
    if dir = "" then ""
    else
        System.IO.Directory.GetFiles(dir, "*.fun")
        |> Array.sort
        |> Array.map System.IO.File.ReadAllText
        |> String.concat "\n"

let combinedSrc = if preludeSrc = "" then src else preludeSrc + "\n" + src
let ast = parseProgram combinedSrc inputPath
```

Change the existing line `let ast = parseProgram src inputPath` to `let ast = parseProgram combinedSrc inputPath`.

NOTE on load order: Alphabetical sort gives: Array.fun, Char.fun, Hashtable.fun, List.fun, Option.fun, Result.fun, String.fun, StringBuilder.fun. Since `prePassDecls` pre-scans ALL decls before elaboration, the Option/Result type definitions will be registered regardless of file order. Alphabetical is fine.
