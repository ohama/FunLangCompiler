```fsharp
module Program

open System

/// Minimal CLI for Phase 1.
/// Usage: langbackend <inputfile> -o <outputbinary>
/// In Phase 1 the input file is IGNORED — always compiles return42Module.
/// Phase 2 introduces real source parsing.
[<EntryPoint>]
let main argv =
    // Parse -o flag: find "-o" and take the next arg as output path
    let args = argv |> Array.toList
    let rec findOutput = function
        | "-o" :: out :: _ -> Some out
        | _ :: rest -> findOutput rest
        | [] -> None

    match findOutput args with
    | None ->
        eprintfn "Usage: langbackend <inputfile> -o <outputbinary>"
        1
    | Some outputPath ->
        // Phase 1: always compile the hardcoded return42Module
        // Phase 2 will parse the input file and elaborate a real AST
        match Pipeline.compile MlirIR.return42Module outputPath with
        | Ok () ->
            0
        | Error (Pipeline.MlirOptFailed (code, err)) ->
            eprintfn "mlir-opt failed (exit %d):\n%s" code err
            1
        | Error (Pipeline.TranslateFailed (code, err)) ->
            eprintfn "mlir-translate failed (exit %d):\n%s" code err
            1
        | Error (Pipeline.ClangFailed (code, err)) ->
            eprintfn "clang failed (exit %d):\n%s" code err
            1
```

Notes on this test:
- `%S` is the directory containing the test file (`tests/compiler/`). `%S/../../src/LangBackend.Cli/LangBackend.Cli.fsproj` resolves to `src/LangBackend.Cli/LangBackend.Cli.fsproj` relative to the repo root.
- `%input` is a temp file FsLit creates containing the `// --- Input:` section (`42\n`). The CLI ignores it in Phase 1 but the pattern is correct for Phase 2+.
- `OUTBIN=$(mktemp ...)` avoids using FsLit's `%output` for the binary (FsLit overwrites `%output` with expected content — using it as a binary path would corrupt the file).
- `$OUTBIN; echo $?` runs the binary and prints its exit code as the last stdout line.
- `// --- Output: 42` matches the exit code.
