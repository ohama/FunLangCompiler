1. **lang_list_sort_by** -- key extractor closure sorts list ascending by extracted int64 key. Uses parallel arrays + insertion sort:
```c
LangCons* lang_list_sort_by(void* closure, LangCons* list) {
    /* Count elements */
    int64_t n = 0;
    LangCons* cur = list;
    while (cur != NULL) { n++; cur = cur->tail; }
    if (n <= 1) return list;

    /* Build parallel arrays: elements and keys */
    int64_t* elems = (int64_t*)GC_malloc((size_t)(n * 8));
    int64_t* keys  = (int64_t*)GC_malloc((size_t)(n * 8));
    LangClosureFn fn = *(LangClosureFn*)closure;
    cur = list;
    for (int64_t i = 0; i < n; i++) {
        elems[i] = cur->head;
        keys[i]  = fn(closure, cur->head);
        cur = cur->tail;
    }

    /* Insertion sort by key (stable) */
    for (int64_t i = 1; i < n; i++) {
        int64_t ke = keys[i], ve = elems[i];
        int64_t j = i - 1;
        while (j >= 0 && keys[j] > ke) {
            keys[j+1]  = keys[j];
            elems[j+1] = elems[j];
            j--;
        }
        keys[j+1]  = ke;
        elems[j+1] = ve;
    }

    /* Reconstruct list from sorted elements (build tail-first) */
    LangCons* head = NULL;
    for (int64_t i = n - 1; i >= 0; i--) {
        LangCons* cell = (LangCons*)GC_malloc(sizeof(LangCons));
        cell->head = elems[i];
        cell->tail = head;
        head = cell;
    }
    return head;
}
```

CRITICAL: This function takes a KEY EXTRACTOR (not comparator). The closure is called once per element with `fn(closure, element)` returning an int64 key. Sort is ascending by key. Do NOT call the closure with two arguments.

CRITICAL: Place this function AFTER the LangCons typedef in lang_runtime.c. The function uses LangClosureFn which is `int64_t (*)(void*, int64_t)` -- same ABI as array_map.
